            </div>
          </div>
          <div class="hero-side" aria-hidden="true"></div>
        </section>

        <section>
          <h3 style="margin:6px 0 12px 0">States</h3>
          <div class="grid" id="state-grid">
            ${Object.keys(DATA).map(k => {

              const img = s.spots && s.spots.length ? s.spots[0].image : s.hero;
              return `<article class="card" role="button" onclick="location.hash='state=${k}'" aria-label="${s.name}">

                <div class="body">
                  <div>
                    <h3>${s.name}</h3>
                    <p>${s.spots.length} featured spots</p>

                </div>
              </article>`;
            }).join('')}
          </div>
        </section>
      `;
    }

    function stateView(stateKey){
      const state = DATA[stateKey];
      if(!state) { location.hash = 'home'; return; }
      app.innerHTML = `
        <div class="crumbs"><a href="#home">States</a> / <strong>${state.name}</strong></div>
        <div class="detail-banner" style="background-image:url('${state.hero}')"></div>
        <h2 style="margin-top:12px">${state.name} — Featured Spots</h2>
        <div class="grid" style="margin-top:12px">
          ${state.spots.map(spot => `
            <article class="card" onclick="location.hash='state=${stateKey}&spot=${spot.id}'" role="button">
              <div class="img" style="background-image:url('${spot.image}')"></div>
              <div class="body">
                <div>
                  <h3>${spot.name}</h3>
                  <p>${spot.short}</p>
                </div>
                <div style="margin-top:12px;color:#587">View Details →</div>
              </div>
            </article>
          `).join('')}
        </div>
      `;
    }

    function spotView(stateKey, spotId){
      const state = DATA[stateKey];
      if(!state){ location.hash='home'; return; }
      const spot = state.spots.find(s=>s.id===spotId);
      if(!spot){ location.hash='state='+stateKey; return; }

      app.innerHTML = `
        <div class="crumbs"><a href="#home">States</a> / <a href="#state=${stateKey}">${state.name}</a> / <strong>${spot.name}</strong></div>
        <div class="detail-banner" style="background-image:url('${spot.image}')"></div>

        <div class="detail-info">
          <div class="detail-text">
            <h2 style="margin-top:0">${spot.name}</h2>
            <p style="color:#465">${spot.description}</p>

            <div style="margin-top:14px">
              <h4 style="margin:6px 0">Best Time to Visit</h4>
              <p style="margin:2px 0 10px 0">${spot.best_time}</p>

              <h4 style="margin:6px 0">Things to Do</h4>
              <ul class="checks">
                ${spot.things_to_do.map(it=>`<li>${it}</li>`).join('')}
              </ul>

              <h4 style="margin:10px 0 6px 0">How to Reach</h4>
              <p style="margin:2px 0">${spot.how_to_reach}</p>
            </div>

            <div style="margin-top:14px">
              <h4 style="margin:6px 0">Gallery</h4>
              <div class="gallery">
                ${spot.gallery.map(src=>`<img src="${src}" alt="${spot.name} image">`).join('')}
              </div>
            </div>
          </div>

          <aside class="meta-box">
            <h4>Quick Info</h4>
            <p><strong>State:</strong> ${state.name}</p>
            <p><strong>Category:</strong> Scenic / Cultural</p>
            <p style="margin-top:10px"><button class="btn" onclick="location.hash='state=${stateKey}'">Back to ${state.name}</button></p>
          </aside>
        </div>
      `;
    }

    function renderFromHash(){
      const hash = location.hash.slice(1);
      if(!hash || hash==='home'){ homeView(); return; }

      const params = new URLSearchParams(hash);
      if(params.has('state') && params.has('spot')){
        const state = params.get('state'), spot = params.get('spot');
        spotView(state, spot);
      } else if(params.has('state')){
        const state = params.get('state');
        stateView(state);
      } else {
        // fallback: check plain format like "state=karnataka&spot=coorg"
        if(hash.startsWith('state=')){
          const p = new URLSearchParams(hash);
          const s = p.get('state');
          const sp = p.get('spot');
          if(sp) spotView(s,sp); else stateView(s);
        } else {
          homeView();
        }
      }
    }

    // small helper used by hero button
    function scrollToGrid(){
      const el = document.getElementById('state-grid');
      if(el) el.scrollIntoView({behavior:'smooth'});
    }

    // init routing
    window.addEventListener('hashchange', renderFromHash);
    window.addEventListener('load', renderFromHash);

    // accessibility: handle keyboard "Enter" for cards rendered as role=button

      // nothing special here yet; cards use onclick
    });
